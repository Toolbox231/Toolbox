        Esp1.Text = "Esp Toilet On"

            if v.Name == "1212" then
                if not v:FindFirstChild("HighlightBoxHandleAdornment") then
                    local HighlightBoxHandleAdornment = Instance.new("BoxHandleAdornment")
                    HighlightBoxHandleAdornment.Name = "HighlightBoxHandleAdornment"
                    HighlightBoxHandleAdornment.Adornee = v
                    HighlightBoxHandleAdornment.Size = Vector3.new(2, 2, 2)
                    HighlightBoxHandleAdornment.AlwaysOnTop = true
                    HighlightBoxHandleAdornment.Color3 = Color3.fromRGB(255, 215, 0) -- золотой цвет
                    HighlightBoxHandleAdornment.Transparency = 0.2
                    HighlightBoxHandleAdornment.ZIndex = 10
                    HighlightBoxHandleAdornment.Parent = v
                end
            end
        end
    else
        Esp1.Text = "Esp Toilet Off"
        -- отключить ESP код
        for i, v in pairs(game:GetService("Workspace").Interactables.MorphTriggers:GetChildren()) do
            if v:FindFirstChild("HighlightBoxHandleAdornment") then
                v:FindFirstChild("HighlightBoxHandleAdornment"):Destroy()

    end
